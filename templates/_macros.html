{%- macro display_date(date) -%} {%- set day = date | date(format="%e") -%} {%- if day is ending_with("1") -%} {%- set suffix = "st" -%} {%- elif day is ending_with("2") -%} {%- set suffix = "nd" -%} {%- elif day is ending_with("3") -%} {%- set suffix = "rd" -%} {%- else -%} {%- set suffix = "th" -%} {%- endif -%} {{ day }}{{ suffix }} {{ date | date(format="%B %Y") }} {%- endmacro nicedate -%}
